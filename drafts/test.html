Matplotlib¶

%matplotlib inline
import matplotlib.pyplot as plt
import numpy as np

arr = np.random.uniform(size=(4, 6))

fig = plt.figure(figsize=(17, 7))

for i in range(1, 3):
    ax = fig.add_subplot(1, 2, i)
    ax.matshow(arr, cmap='viridis')
    
plt.tight_layout()
